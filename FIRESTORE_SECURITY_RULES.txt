rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Rhapsody/Devotionals Collection
    match /rhapsody/{document=**} {
      // Read: Anyone can read devotionals
      allow read: if true;
      
      // Write (create, update, delete): Only authenticated users (admins) can write
      allow write: if request.auth != null;
      
      // Additional security: Only creators can update/delete their own posts
      allow update, delete: if request.auth.uid == resource.data.createdBy || request.auth != null;
    }
    
    // Events Collection
    match /events/{document=**} {
      // Read: Anyone can read events
      allow read: if true;
      
      // Write: Only authenticated users (admins) can create/update/delete
      allow write: if request.auth != null;
    }
    
    // Groups/Cells Collection
    match /groups/{document=**} {
      // Read: Anyone can read groups
      allow read: if true;
      
      // Write: Only authenticated users can join/create
      allow write: if request.auth != null;
    }
    
    // Users Collection (if needed)
    match /users/{uid} {
      // Users can only read/write their own document
      allow read, write: if request.auth.uid == uid;
      
      // Admins can read all users
      allow read: if request.auth != null && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Deny all other access by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
